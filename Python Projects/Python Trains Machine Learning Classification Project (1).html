df['Arrival Time DC'] = pd.to_datetime(df['Arrival Time'], format='%H:%M:%S', errors='coerce')

    df['Departure Time DC'].notna() & df['Arrival Time DC'].notna(),
    (df['Arrival Time DC'] - df['Departure Time DC']).dt.total_seconds() / 3600,
    np.nan
)

#Creating a confusion matrix, accuracy and classification report for the logistic regression model. For the model we are choosing less than the average to classify as low price.

THRESHOLD = df['Price'].mean()
print(f"Average Ticket Pricing: {THRESHOLD}")
df['Low Price'] = (df['Price'] < THRESHOLD).astype(int)
X= pc_df[existing_pcs]
yy= df['Low Price']

XX_train, XX_test, yy_train, yy_test = train_test_split(X, yy, test_size=0.3, random_state=42)

clf = LogisticRegression(max_iter=1000)
clf.fit(XX_train, yy_train)
yy_pred = clf.predict(XX_test)
print("Accuracy:", accuracy_score(yy_test, yy_pred))
print("\nConfusion Matrix:\n", confusion_matrix(yy_test, yy_pred))
print("\nClassification Report:\n", classification_report(yy_test, yy_pred, target_names=['Not Low Price', 'Low Price']))
